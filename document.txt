@client_bp.route('/cliente/cadastrar', methods=['GET', 'POST'])
def client_register_fjpj():
    # ✅ Usa o formulário renomeado
    form = FormClientPFJ()
    
    if form.validate_on_submit():
        try:
            # 1. Determinar o tipo (Física 'F' ou Jurídica 'J')
            code_data = ''.join(filter(str.isdigit, form.code.data))
            client_type = 'F' if len(code_data) == 11 else 'J'
            
            # 2. Armazenar dados básicos na session para o próximo passo
            session['client_reg_data'] = { # ✅ Usa chave mais clara
                'code': code_data, 
                'name': form.name.data,
                'contact': form.contact.data, 
                'email': form.email.data,
                'type': client_type
            }
            
            # 3. Redirecionar para o passo 2 (Endereço)
            return redirect(url_for('client_bp.client_register_address'))

        except Exception as e:
            flash(f'Erro interno ao processar dados. {e}', 'danger')
            
    # ✅ Renderiza o template de cadastro inicial (NOME PROFISSIONALIZADO)
    return render_template('client/client_create_personal_data.html', form=form, titulo="Registrar Novo Cliente", stperson="active")

 @nfe_bp.route('/gerar-xml-nfce/<int:id>')
def gerar_xml_nfce(id):

    invoice = Invoice.query.get_or_404(id)


    if not invoice.number or invoice.number > 999999999:
        raise Exception("Número da NF inválido (nNF deve ter até 9 dígitos)")

    client = invoice.client
    items = invoice.items

    # 🔐 Dados do emitente via STORE (sessão)
    store = session.get("Store")
    if not store:
        raise Exception("Dados da loja (STORE) não encontrados na sessão")

    emitente = {
        "code": store["Code"],
        "name": store["Name"],
        "state_registration": store["State_Registration"],
        "crt": store["CRT"],
        "address": {
            "xLgr": store["Address"],
            "nro": store["Number"],
            "xBairro": store["Neighborhood"],
            "cMun": IBGE_CITY_CODE,      # config.py
            "xMun": store["City"],
            "UF": store["UF"],
            "CEP": store["ZipCode"],
            "cPais": "1058",
            "xPais": "Brasil"
        }
    }

    # 🔗 QRCode (se já existir função)
    digest_value = ""  # ainda vazio antes da assinatura
    qrcode_url = gerar_qrcode_url(
        chave=invoice.access_key,
        valor_nf=invoice.total_value,
        digest_value=digest_value
    )


    # 🧾 Geração do XML
    xml_bytes = gerar_xml_nfce(
        invoice=invoice,
        client=client,
        items=items,
        emitente=emitente,
        qrcode_url=qrcode_url,
        tp_amb=TP_AMB
    )

    # 💾 Salva XML
    xml_name = f"NFCe_{invoice.access_key}.xml"
    xml_path = f"admin/nfe/output/{xml_name}"

    with open(xml_path, "wb") as f:
        f.write(xml_bytes)

    invoice.xml_path = xml_path
    invoice.status = "X"
    db.session.commit()

    flash("Geração do XML NFC-e concluída com sucesso!", "success")
    return redirect(url_for("nfe_bp.nfe_list"))

-- Tabela de Posts do Blog
CREATE TABLE IF NOT EXISTS ouvirtiba.blog_post (
    id SERIAL PRIMARY KEY,
    store_id INTEGER NOT NULL,
    title VARCHAR(255) NOT NULL,
    summary VARCHAR(355) NOT NULL,
    content TEXT NOT NULL,
    author VARCHAR(45) NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP,
    image VARCHAR(150),
    active BOOLEAN DEFAULT TRUE,
    slug VARCHAR(255) UNIQUE,
    
    -- Chave estrangeira (assumindo que existe uma tabela 'store')
    CONSTRAINT fk_store FOREIGN KEY (store_id) REFERENCES ouvirtiba.store(id) ON DELETE CASCADE
);

-- Índices para melhorar performance
CREATE INDEX idx_blog_post_store_id ON ouvirtiba.blog_post(store_id);
CREATE INDEX idx_blog_post_created_at ON ouvirtiba.blog_post(created_at DESC);
CREATE INDEX idx_blog_post_active ON ouvirtiba.blog_post(active);
CREATE INDEX idx_blog_post_slug ON ouvirtiba.blog_post(slug);

-- Comentários para documentação
COMMENT ON TABLE ouvirtiba.blog_post IS 'Tabela de posts do blog';
COMMENT ON COLUMN ouvirtiba.blog_post.slug IS 'URL amigável gerada a partir do título';
COMMENT ON COLUMN ouvirtiba.blog_post.active IS 'Indica se o post está publicado ou rascunho';
